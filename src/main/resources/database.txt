create table articulos_particular(
    numero_serie varchar(30) primary key,
    modelo varchar(30) foreign key references modelos_general(modelo),
    ubicacion varchar(30),
    estado varchar(30),
    plazo_Garantia varchar(30),
    fecha_Ingreso Date,
    fecha_Ultimo_Mantenimiento Date,
    fecha_Proximo_Mantenimiento Date,
    cantidad int,
    constraint CK_articulo_fechas check (fecha_Ingreso <= fecha_Ultimo_Mantenimiento AND fecha_Ultimo_Mantenimiento < fecha_Proximo_Mantenimiento)
)

create table asignacion_practica(
    idasignacion int primary key identity,
    idpractica int foreign key references practicas(idpractica),
    clave_grupo varchar(4) foreign key references grupos(clave_grupo),
    fecha date,
    estado varchar(30),
    idlugar int foreign key references lugar(idlugar)
    constraint UQ_asignacion unique(idpractica,clave_grupo)
)

create table avance_curricular(
    IDAvance int primary key identity,
    IDPractica int foreign key references practicas(IDPractica),
    clave_grupo varchar(4) foreign key references grupos(clave_grupo),
    fecha date,
    estado varchar(30),
)

create table expediente_articulo(
	IDExpediente int identity primary key,
	numero_serie varchar(50) foreign key references articulos_particular,
	numero_control varchar(10) foreign key references usuarios(numero_control),
	fecha_entrada date,
	fecha_salida date,
)

create table expediente_mesa(
    IDExpediente int identity primary key,
    IDLugar int foreign key lugar(IDLugar),
    numero_control varchar(10) foreign key references usuarios(numero_control),
    fecha_entrada date,
    fecha_salida date,
    fecha_proximo_mantenimiento date
)

create table grupos(
    clave_grupo varchar(4) primary key,
    clave_materia varchar(4) foreign key references materias(clave_Materia),
    numero_control varchar(15) foreign key references usuarios(numero_Control),
    salon varchar(50),
    hora_entrada time,
    hora_salida time,
    dias_laboratorio varchar(100),
    constraint CK_grupos_horario_valido check (hora_entrada <= hora_salida),
    constraint UQ_grupos_salon_horario unique (salon, hora_entrada, hora_salida, dias_laboratorio)
)

create table insumos_practicas(
    IDInsumos int primary key identity,
    numero_serie varchar(50) foreign key references articulos_particular(numero_serie),
    IDAsignacion int foreign key references asignacion_practica(idasignacion),
    estado varchar(30),
)

create table lugar(
    IDLugar int identity primary key,
    nombre varchar(50),
    estado varchar(50),
    fecha_proximo_mantenimiento date,
    fecha_ultimo_mantenimiento date,
    fecha_salida_mantenimiento date,
    nombre_personal varchar(50)
)

create table mantenimiento_articulo(
    IDMantenimiento int identity primary key,
    numero_serie int foreign key references articulos_particular(numero_serie),
    numero_control int foreign key references usuarios(numero_control),
    motivo varchar(100),
    estado varchar(50),
    fecha_Entrada date,
    fecha_Salida date,
    fecha_Proximo_Mantenimiento date
)

create table mantenimiento_mesa(
    IDMantenimiento int identity,
    IDLugar int foreign key references lugar(IDLugar),
    numero_control varchar(10) foreign key references usuarios(numero_control)
    fecha_Entrada date,
    fecha_Salida date,
    fecha_Proximo_mantenimiento date,
)

create table materias(
    clave_materia varchar(4) primary key,
    nombre_materia varchar(100)
)

create table modelos_general(
    modelo varchar(50) primary key,
    descripcion varchar(MAX),
    marca varchar(30),
    categoria varchar(30)
)

create table practicas(
    idpractica int primary key identity,
    clave_materia varchar(4) foreign key references materias(clave_materia),
    nombre varchar(100),
    archivo varchar(max),
)

create table usuarios(
    numero_Control varchar(15) primary key,
    nombre varchar(100),
    correo varchar(50),
    telefono varchar(15),
    password_hash varchar(255),
    rol varchar(30),
    reset_token varchar(255)
)

create procedure SP_CheckDisponibilidad
    @fecha date,
    @HoraEntrada time,
    @HoraSalida time,
    @IDLugar int = null
AS
BEGIN
    SET NOCOUNT ON;

    SELECT
        l.idlugar,
        l.nombre,
        l.estado,
        l.fecha_ultimo_mantenimiento,
        l.fecha_proximo_mantenimiento,
        l.fecha_salida_mantenimiento,
        l.nombre_personal
    FROM lugar l
    WHERE l.estado = 'Disponible'
      and (l.fecha_proximo_mantenimiento >= DATEADD(DAY, 7,@fecha))
      AND (l.nombre LIKE '%Salon%'
          OR NOT EXISTS (
              SELECT 1
              FROM asignacion_practica ap
              INNER JOIN grupos g ON ap.clave_grupo = g.clave_grupo
              WHERE ap.idlugar = l.idlugar
                AND ap.fecha = @fecha
                AND @HoraEntrada < g.hora_salida
                AND @HoraSalida > g.hora_entrada
          )
      )
      or l.IDLugar = @IDLugar
END

create procedure SP_InsertarGrupo
    @salon NVARCHAR(50),
    @horaEntrada TIME,
    @horaSalida TIME,
    @DiasLaboratorio VARCHAR(100)
AS
BEGIN
    SET NOCOUNT ON;
    SELECT CASE WHEN EXISTS (
        SELECT 1
        FROM grupos g
        CROSS APPLY STRING_SPLIT(g.dias_laboratorio, ',') gd
        CROSS APPLY STRING_SPLIT(@DiasLaboratorio, ',') nd
        WHERE g.salon = @salon
            AND gd.value = nd.value
            AND (g.hora_entrada < @horaSalida AND g.hora_salida > @horaEntrada))
        THEN 1 ELSE 0
    END AS Existe;
END;

create procedure SP_ListarArticulos
as begin
select
    m.modelo as modelo,
    m.descripcion as descripcion,
    m.categoria as categoria,
    m.marca as marca,

    a.numero_serie as numeroSerie,
    a.estado as estado,
    a.plazo_Garantia as plazoGarantia,
    a.ubicacion as ubicacion,

    a.fecha_Ingreso as fechaIngreso,
    a.fecha_Ultimo_Mantenimiento as fechaUltimoMantenimiento,
    a.fecha_Proximo_Mantenimiento as fechaProximoMantenimiento
    from articulos_particular a
    inner join modelos_general m
        on a.modelo=m.modelo
end

create procedure SP_ListarEquiposMantenimientos
AS
BEGIN
    SELECT
        em.IDMantenimiento,
        ap.numero_serie,
        mg.modelo,
        mg.descripcion,
        mg.marca,
        em.motivo,

        u.nombre,
        u.numero_control,
        em.fecha_Entrada,
        em.fecha_Salida,
        em.estado
    FROM mantenimiento_articulo em
    INNER JOIN articulos_particular ap
        ON em.articulo = ap.numero_serie
    INNER JOIN modelos_general mg
        ON ap.modelo = mg.modelo
    INNER JOIN usuarios u
        ON em.numero_control = u.numero_control
END

create procedure SP_ListarGrupos @Salon varchar(30)
as begin
    select
    m.clave_materia,
    m.nombre_materia,
    g.clave_grupo,
    u.numero_Control,
    u.nombre,
    g.hora_entrada,
    g.hora_salida,
    g.dias_laboratorio
    from grupos g
    inner join materias m
        on g.clave_materia = m.clave_materia
    inner join usuarios u
        on g.numero_control = u.numero_control
    where g.salon = @Salon
end

create procedure SP_ListarInsumos
    @fechaInicio DATE,
    @fechaTermino DATE
AS
BEGIN
    SELECT
        m.modelo,
        m.descripcion,
        m.categoria,
        m.marca,
        a.numero_serie,
        a.estado,
        a.ubicacion,
        a.fecha_Ultimo_Mantenimiento
    FROM articulos_particular a
    INNER JOIN modelos_general m
        ON a.modelo = m.modelo
    WHERE
        a.estado = 'Disponible'
        AND a.fecha_Proximo_Mantenimiento NOT BETWEEN @fechaInicio AND @fechaTermino
END

create procedure SP_ListarLugares
as select l.IDLugar,
          l.nombre,
          l.estado,
          l.fecha_ultimo_mantenimiento,
          l.fecha_proximo_mantenimiento,
          m.fecha_salida
          from lugar l
          inner join mantenimiento_mesa m
            on m.IDLugar = l.IDLugar

create procedure SP_ListarUsuario
as select numero_Control, nombre, correo, telefono, rol from usuarios

create procedure SP_UpdateGrupo
    @salon NVARCHAR(50),
    @horaEntrada TIME,
    @horaSalida TIME,
    @DiasLaboratorio VARCHAR(100),
    @ClaveGrupo varchar(10) AS
BEGIN
    SET NOCOUNT ON;
    SELECT CASE WHEN EXISTS (
        SELECT 1 FROM grupos g
        CROSS APPLY STRING_SPLIT(g.dias_laboratorio, ',') gd
        CROSS APPLY STRING_SPLIT(@DiasLaboratorio, ',') nd
        WHERE g.salon = @salon
            AND gd.value = nd.value
            AND (g.hora_entrada < @horaSalida AND g.hora_salida > @horaEntrada)
            and g.clave_grupo <> @ClaveGrupo)
        THEN 1 ELSE 0
    END AS Existe;
END;
